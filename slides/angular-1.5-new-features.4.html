Multi slot transclusion

Map transcluded content to specified slots.
Like shadow dom Content Projection. 

<pane>
    <pane-title>
        <a ng-click="...">a title</a>
    <pane-title>
    <pane-content>
        The content of the pane
    </pane-content>
</pane>

mod.component('pane', {
    restrict: 'E',
    transclude: { paneTitle: '?titleSlot',
                  paneContent: 'contentSlot' },
    template:
        '<h1 ng-transclude="titleSlot"></h1>' +
        <div ng-transclude="contentSlot"></div>'
});
